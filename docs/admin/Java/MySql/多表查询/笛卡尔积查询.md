## 笛卡尔积查询

所有多表查询的，最基础查询；笛卡尔积查询本身没有意义，平时不会用。

```sql
-- 格式
create table a(
    id int,
    name varchar(20)
);

create table b(
    id int,
    name varchar(20)
);

select * from a, b;
```

```sql
create table a(
    id int,
    name varchar(20)
);
insert into a values(1,'a');
insert into a values(2,'b');
insert into a values(3,'c');

create table b(
    id int,
    name varchar(20)
);
insert into b values(1,'aaa');
insert into b values(2,'bbb');
insert into b values(4,'ccc');

select * from a, b;
-- 结果:
-- 1 a 1 aaa
-- 2 b 1 aaa
-- 3 c 1 aaa
-- 1 a 2 bbb
-- 2 b 2 bbb
-- 3 c 2 bbb
-- 1 a 4 ccc
-- 2 b 4 ccc
-- 3 c 4 ccc
```
