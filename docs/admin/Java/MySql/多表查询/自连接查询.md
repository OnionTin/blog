## 自连接查询

```sql
-- 自外连接查询格式
select * from a a1,a a2 where a1.id = a2.id;
```

```sql
-- 自连接查询案例
create table a(
    id int,
    name varchar(20)
);
insert into a values(1,'a');
insert into a values(2,'b');
insert into a values(3,'c');

select * from a a1,a a2 where a1.id = a2.id;
-- 结果:
-- +----+------+----+------+
-- | id | name | id | name |
-- +----+------+----+------+
-- |  1 | a    |  1 | a    |
-- |  2 | b    |  2 | b    |
-- |  3 | c    |  3 | c    |
-- +----+------+----+------+
```
